## 1. 环境准备

### 1.1 系统要求

- Windows 10/11 操作系统
- Visual Studio 2022 或更高版本
- .NET Framework 4.8 开发工具

### 1.2 项目引用

1. 在 Visual Studio 中创建 **Windows Forms 应用（.NET Framework 4.8）** 项目。  
2. 在解决方案中添加对 **`IndustrialControls`** 的**项目引用**（指向本仓库 `src/IndustrialControls/IndustrialControls.csproj`），或引用已编译的 **`IndustrialControls.dll`**（需满足相同依赖，详见使用手册）。  
3. 生成宿主项目后，在代码中 `using` 相应命名空间；控件可从工具箱添加或按使用手册说明注册。

## 2. 创建第一个应用

### 2.1 状态指示器

```csharp
using IndustrialControls.Controls.StatusIndicator;

// 创建状态指示器
var indicator = new StatusIndicator
{
    State = IndicatorState.Running,    // 运行状态
    Label = "电机1",                    // 标签文本
    EnableBlink = true,                // 启用闪烁
    BlinkInterval = 500,               // 闪烁间隔 500ms
    Shape = IndicatorShape.Circle,     // 圆形外观
    Size = new Size(60, 80)
};

// 设置位置
indicator.Location = new Point(20, 20);

// 添加到窗体
this.Controls.Add(indicator);

// 监听状态变化事件
indicator.StateChanged += (s, state) =>
{
    Console.WriteLine($"状态变更: {state}");
};

// 切换状态
indicator.State = IndicatorState.Fault;  // 故障状态
```

### 2.2 设备控制按钮

```csharp
using IndustrialControls.Controls.DeviceButton;

// 创建设备控制按钮
var btn = new DeviceControlButton
{
    Size = new Size(120, 40),
    Location = new Point(20, 120),
    EnableLongPress = true,     // 启用长按确认
    LongPressTime = 1000        // 长按时间 1秒
};

// 设置多状态
btn.SetStates(
    new DeviceButtonState { DisplayText = "运行", Color = Color.Green },
    new DeviceButtonState { DisplayText = "停止", Color = Color.Red }
);

// 添加到窗体
this.Controls.Add(btn);

// 监听状态变化
btn.StateChanged += (s, state) =>
{
    MessageBox.Show($"当前状态: {state.DisplayText}");
};
```

### 2.3 报警显示

```csharp
using IndustrialControls.Controls.Alarm;

// 创建报警显示控件
var alarmDisplay = new AlarmDisplay
{
    Size = new Size(400, 300),
    Location = new Point(20, 180),
    AutoScroll = true           // 自动滚动到最新报警
};

// 添加到窗体
this.Controls.Add(alarmDisplay);

// 添加报警
alarmDisplay.AddAlarm(AlarmLevel.Emergency, "紧急报警：电机过载", "设备A");
alarmDisplay.AddAlarm(AlarmLevel.Important, "重要报警：温度过高", "设备B");
alarmDisplay.AddAlarm(AlarmLevel.General, "一般报警：压力异常", "设备C");

// 确认报警
alarmDisplay.AcknowledgeAll();
```

## 3. 参数持久化

### 3.1 使用 ParameterManager

```csharp
using IndustrialControls.Core;

// 创建参数管理器实例
var paramManager = new ParameterManager("config/appsettings.json");

// 设置参数
paramManager.SetValue("Communication", "ServerIP", "192.168.1.100");
paramManager.SetValue("Communication", "Port", 8080);
paramManager.SetValue("System", "AutoStart", true);
paramManager.SetValue("System", "Timeout", 3000);

// 获取参数（带默认值）
string ip = paramManager.GetValue("Communication", "ServerIP", "127.0.0.1");
int port = paramManager.GetValue("Communication", "Port", 80);
bool autoStart = paramManager.GetValue("System", "AutoStart", false);

// 监听参数变更
paramManager.ParameterChanged += (s, e) =>
{
    Console.WriteLine($"参数变更: {e.GroupName}.{e.Key} = {e.Value}");
};
```

## 4. 主题切换

### 4.1 使用主题管理器

```csharp
using IndustrialControls.Theme;

// 切换到暗色主题
ThemeManager.Instance.ApplyTheme<FlatDarkTheme>();

// 切换到亮色主题
ThemeManager.Instance.ApplyTheme<FlatLightTheme>();

// 获取当前主题颜色
var colors = ThemeManager.Instance.CurrentTheme.Colors;
btn.BackColor = colors.Primary;
btn.ForeColor = colors.TextOnPrimary;
```

## 5. 通信控制

### 5.1 配置通信控件

```csharp
using IndustrialControls.Controls.Communication;

// 创建通信控制控件
var commControl = new CommunicationControl
{
    Size = new Size(300, 150),
    Location = new Point(20, 500),
    CommunicationMode = CommunicationMode.TcpClient,
    Host = "192.168.1.100",
    Port = 502
};

// 添加到窗体
this.Controls.Add(commControl);

// 连接
commControl.Connect();

// 发送数据
commControl.SendData(new byte[] { 0x01, 0x02, 0x03 });

// 监听接收到的数据
commControl.DataReceived += (s, data) =>
{
    string hex = BitConverter.ToString(data);
    Console.WriteLine($"收到数据: {hex}");
};
```

## 6. 虚拟键盘

### 6.1 启用焦点跟踪并显示键盘窗体

```csharp
using IndustrialControls.Controls.VirtualKeyboard;

// 在程序启动时调用一次：轮询当前焦点输入控件（供键盘将字符写回）
VirtualKeyboardManager.Initialize();

var textBox = new TextBox { Location = new Point(20, 680) };
this.Controls.Add(textBox);

var vk = new VirtualKeyboardForm();
vk.SetTargetControl(textBox);
vk.KeyInput += (s, e) => { /* 根据 e.Key / e.Target 写入控件，或参考 Demo 中 VirtualKeyboardPage */ };
vk.ShowAt(textBox.PointToScreen(new Point(0, textBox.Height + 4)));
```

## 7. 完整示例

```csharp
using System;
using System.Drawing;
using System.Windows.Forms;
using IndustrialControls.Controls.StatusIndicator;
using IndustrialControls.Controls.DeviceButton;
using IndustrialControls.Controls.Alarm;
using IndustrialControls.Core;

namespace IndustrialControls.Demo
{
    public partial class MainForm : Form
    {
        private ParameterManager _paramManager;
        private StatusIndicator _indicator;
        private DeviceControlButton _controlBtn;
        private AlarmDisplay _alarmDisplay;

        public MainForm()
        {
            InitializeComponent();
            InitializeControls();
            LoadParameters();
        }

        private void InitializeControls()
        {
            // 状态指示器
            _indicator = new StatusIndicator
            {
                Location = new Point(20, 20),
                Size = new Size(60, 80),
                State = IndicatorState.Stopped,
                Label = "主电机",
                EnableBlink = true
            };
            this.Controls.Add(_indicator);

            // 控制按钮
            _controlBtn = new DeviceControlButton
            {
                Location = new Point(100, 20),
                Size = new Size(120, 40),
                EnableLongPress = true
            };
            _controlBtn.SetStates(
                new DeviceButtonState("启动", Color.Green),
                new DeviceButtonState("停止", Color.Red)
            );
            _controlBtn.StateChanged += ControlBtn_StateChanged;
            this.Controls.Add(_controlBtn);

            // 报警显示
            _alarmDisplay = new AlarmDisplay
            {
                Location = new Point(20, 120),
                Size = new Size(400, 200)
            };
            this.Controls.Add(_alarmDisplay);
        }

        private void LoadParameters()
        {
            _paramManager = new ParameterManager("config/demo.json");
            
            // 读取配置
            bool autoStart = _paramManager.GetValue("System", "AutoStart", false);
            if (autoStart)
            {
                _indicator.State = IndicatorState.Running;
                _alarmDisplay.AddAlarm(AlarmLevel.Info, "系统已自动启动", "MainForm");
            }
        }

        private void ControlBtn_StateChanged(object sender, DeviceButtonState e)
        {
            if (e.DisplayText == "启动")
            {
                _indicator.State = IndicatorState.Running;
                _alarmDisplay.AddAlarm(AlarmLevel.Info, "主电机已启动", "MainForm");
            }
            else
            {
                _indicator.State = IndicatorState.Stopped;
                _alarmDisplay.AddAlarm(AlarmLevel.Info, "主电机已停止", "MainForm");
            }
        }
    }
}
```
